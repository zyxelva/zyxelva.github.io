    public TreeNode sortedListToBST(ListNode head) {
        //null
        if (null == head) {
            return null;
        }
        //single node
        if (null == head.next) {

        ListNode rightPart = slow.next.next;

class Solution {
    ListNode newNode;

    public TreeNode sortedListToBST(ListNode head) {
        //null
        if (null == head) {
            return null;
        }
        //single node
        if (null == head.next) {
            return new TreeNode(head.val);
        }
        int n = 0;
        newNode = head;
        while (null != head) {
            head = head.next;
            n++;
        }
        return build(0, n - 1);
    }

    private TreeNode build(int left, int right) {
        if (left > right) {
            return null;
        }
        //middle
        int mid = left + (right - left) / 2;
        //inorder traversal
        //left tree
        TreeNode leftTree = build(left, mid - 1);
        //rootï¼Œæ­¤æ—¶ï¼Œå†æ„é€ å®Œå·¦å­æ ‘åï¼ŒnewNodeå·²ç»æ˜¯é“¾è¡¨çš„ä¸­ç‚¹èŠ‚ç‚¹äº†
        TreeNode root = new TreeNode(newNode.val);
        newNode = newNode.next;
        root.left = leftTree;
        root.right = build(mid + 1, right);
        return root;
    }
}
